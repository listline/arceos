62    /// Copies page table mappings from another address space.
63    ///
64    /// It copies the page table entries only rather than the memory regions,
65    /// usually used to copy a portion of the kernel space mapping to the
66    /// user space.
67    ///
68    /// Note that on dropping, the copied PTEs will also be cleared, which could
69    /// taint the original page table. For workaround, you can use
70    /// [`AddrSpace::clear_mappings`].
71    ///
72    /// Returns an error if the two address spaces overlap.
73    pub fn copy_mappings_from(&mut self, other: &AddrSpace) -> AxResult {
74        if self.va_range.overlaps(other.va_range) {
75            return ax_err!(InvalidInput, "address space overlap");
76        }
77        self.pt.copy_from(&other.pt, other.base(), other.size());
78        Ok(())
79    }

88    fn validate_region(&self, start: VirtAddr, size: usize) -> AxResult {
89        if !self.contains_range(start, size) {
90            return ax_err!(InvalidInput, "address out of range");
91        }
92        if !start.is_aligned_4k() || !is_aligned_4k(size) {
93            return ax_err!(InvalidInput, "address not aligned");
94        }
95        Ok(())
96    }

112    /// Add a new linear mapping.
113    ///
114    /// See [`Backend`] for more details about the mapping backends.
115    ///
116    /// The `flags` parameter indicates the mapping permissions and attributes.
117    ///
118    /// Returns an error if the address range is out of the address space or not
119    /// aligned.
120    pub fn map_linear(
121        &mut self,
122        start_vaddr: VirtAddr,
123        start_paddr: PhysAddr,
124        size: usize,
125        flags: MappingFlags,
126    ) -> AxResult {
127        self.validate_region(start_vaddr, size)?;
128        if !start_paddr.is_aligned_4k() {
129            return ax_err!(InvalidInput, "address not aligned");
130        }
131
132        let offset = start_vaddr.as_usize() - start_paddr.as_usize();
133        let area = MemoryArea::new(start_vaddr, size, flags, Backend::new_linear(offset));
134        self.areas
135            .map(area, &mut self.pt, false)
136            .map_err(mapping_err_to_ax_err)?;
137        Ok(())
138    }

164    /// Populates the area with physical frames, returning false if the area
165    /// contains unmapped area.
166    pub fn populate_area(&mut self, mut start: VirtAddr, size: usize) -> AxResult {
167        self.validate_region(start, size)?;
168        let end = start + size;
169
170        while let Some(area) = self.areas.find(start) {
171            let backend = area.backend();
172            if let Backend::Alloc { populate } = backend {
173                if !*populate {
174                    for addr in PageIter4K::new(start, area.end().min(end)).unwrap() {
175                        match self.pt.query(addr) {
176                            Ok(_) => {}
177                            // If the page is not mapped, try map it.
178                            Err(PagingError::NotMapped) => {
179                                if !backend.handle_page_fault(addr, area.flags(), &mut self.pt) {
180                                    return Err(AxError::NoMemory);
181                                }
182                            }
183                            Err(_) => return Err(AxError::BadAddress),
184                        };
185                    }
186                }
187            }
188            start = area.end();
189            assert!(start.is_aligned_4k());
190            if start >= end {
191                break;
192            }
193        }
194
195        if start < end {
196            // If the area is not fully mapped, we return ENOMEM.
197            return ax_err!(NoMemory);
198        }
199
200        Ok(())
201    }

216    /// To remove user area mappings from address space.
217    pub fn unmap_user_areas(&mut self) -> AxResult {
218        for area in self.areas.iter() {
219            assert!(area.start().is_aligned_4k());
220            assert!(area.size() % PAGE_SIZE_4K == 0);
221            assert!(area.flags().contains(MappingFlags::USER));
222            assert!(
223                self.va_range
224                    .contains_range(VirtAddrRange::from_start_size(area.start(), area.size())),
225                "MemorySet contains out-of-va-range area"
226            );
227        }
228        self.areas.clear(&mut self.pt).unwrap();
229        Ok(())
230    }

232    /// To process data in this area with the given function.
233    ///
234    /// Now it supports reading and writing data in the given interval.
235    ///
236    /// # Arguments
237    /// - `start`: The start virtual address to process.
238    /// - `size`: The size of the data to process.
239    /// - `f`: The function to process the data, whose arguments are the start virtual address,
240    ///   the offset and the size of the data.
241    ///
242    /// # Notes
243    ///   The caller must ensure that the permission of the operation is allowed.
244    fn process_area_data<F>(&self, start: VirtAddr, size: usize, f: F) -> AxResult
245    where
246        F: FnMut(VirtAddr, usize, usize),
247    {
248        Self::process_area_data_with_page_table(&self.pt, &self.va_range, start, size, f)
249    }
250
251    fn process_area_data_with_page_table<F>(
252        pt: &PageTable,
253        va_range: &VirtAddrRange,
254        start: VirtAddr,
255        size: usize,
256        mut f: F,
257    ) -> AxResult
258    where
259        F: FnMut(VirtAddr, usize, usize),
260    {
261        if !va_range.contains_range(VirtAddrRange::from_start_size(start, size)) {
262            return ax_err!(InvalidInput, "address out of range");
263        }
264        let mut cnt = 0;
265        // If start is aligned to 4K, start_align_down will be equal to start_align_up.
266        let end_align_up = (start + size).align_up_4k();
267        for vaddr in PageIter4K::new(start.align_down_4k(), end_align_up)
268            .expect("Failed to create page iterator")
269        {
270            let (mut paddr, _, _) = pt.query(vaddr).map_err(|_| AxError::BadAddress)?;
271
272            let mut copy_size = (size - cnt).min(PAGE_SIZE_4K);
273
274            if copy_size == 0 {
275                break;
276            }
277            if vaddr == start.align_down_4k() && start.align_offset_4k() != 0 {
278                let align_offset = start.align_offset_4k();
279                copy_size = copy_size.min(PAGE_SIZE_4K - align_offset);
280                paddr += align_offset;
281            }
282            f(phys_to_virt(paddr), cnt, copy_size);
283            cnt += copy_size;
284        }
285        Ok(())
286    }

288    /// To read data from the address space.
289    ///
290    /// # Arguments
291    ///
292    /// * `start` - The start virtual address to read.
293    /// * `buf` - The buffer to store the data.
294    pub fn read(&self, start: VirtAddr, buf: &mut [u8]) -> AxResult {
295        self.process_area_data(start, buf.len(), |src, offset, read_size| unsafe {
296            core::ptr::copy_nonoverlapping(src.as_ptr(), buf.as_mut_ptr().add(offset), read_size);
297        })
298    }

332    /// Checks whether an access to the specified memory region is valid.
333    ///
334    /// Returns `true` if the memory region given by `range` is all mapped and
335    /// has proper permission flags (i.e. containing `access_flags`).
336    pub fn check_region_access(
337        &self,
338        mut range: VirtAddrRange,
339        access_flags: MappingFlags,
340    ) -> bool {
341        for area in self.areas.iter() {
342            if area.end() <= range.start {
343                continue;
344            }
345            if area.start() > range.start {
346                return false;
347            }
348
349            // This area overlaps with the memory region
350            if !area.flags().contains(access_flags) {
351                return false;
352            }
353
354            range.start = area.end();
355            if range.is_empty() {
356                return true;
357            }
358        }
359
360        false
361    }

363    /// Handles a page fault at the given address.
364    ///
365    /// `access_flags` indicates the access type that caused the page fault.
366    ///
367    /// Returns `true` if the page fault is handled successfully (not a real
368    /// fault).
369    pub fn handle_page_fault(&mut self, vaddr: VirtAddr, access_flags: MappingFlags) -> bool {
370        if !self.va_range.contains(vaddr) {
371            return false;
372        }
373        if let Some(area) = self.areas.find(vaddr) {
374            let orig_flags = area.flags();
375            if orig_flags.contains(access_flags) {
376                return area
377                    .backend()
378                    .handle_page_fault(vaddr, orig_flags, &mut self.pt);
379            }
380        }
381        false
382    }

384    /// Clone a [`AddrSpace`] by re-mapping all [`MemoryArea`]s in a new page table and copying data in user space.
385    pub fn clone_or_err(&mut self) -> AxResult<Self> {
386        let mut new_aspace = Self::new_empty(self.base(), self.size())?;
387
388        for area in self.areas.iter() {
389            let backend = area.backend();
390            // Remap the memory area in the new address space.
391            let new_area =
392                MemoryArea::new(area.start(), area.size(), area.flags(), backend.clone());
393            new_aspace
394                .areas
395                .map(new_area, &mut new_aspace.pt, false)
396                .map_err(mapping_err_to_ax_err)?;
397
398            if matches!(backend, Backend::Linear { .. }) {
399                continue;
400            }
401            // Copy data from old memory area to new memory area.
402            for vaddr in
403                PageIter4K::new(area.start(), area.end()).expect("Failed to create page iterator")
404            {
405                let addr = match self.pt.query(vaddr) {
406                    Ok((paddr, _, _)) => paddr,
407                    // If the page is not mapped, skip it.
408                    Err(PagingError::NotMapped) => continue,
409                    Err(_) => return Err(AxError::BadAddress),
410                };
411                let new_addr = match new_aspace.pt.query(vaddr) {
412                    Ok((paddr, _, _)) => paddr,
413                    // If the page is not mapped, try map it.
414                    Err(PagingError::NotMapped) => {
415                        if !backend.handle_page_fault(vaddr, area.flags(), &mut new_aspace.pt) {
416                            return Err(AxError::NoMemory);
417                        }
418                        match new_aspace.pt.query(vaddr) {
419                            Ok((paddr, _, _)) => paddr,
420                            Err(_) => return Err(AxError::BadAddress),
421                        }
422                    }
423                    Err(_) => return Err(AxError::BadAddress),
424                };
425                unsafe {
426                    core::ptr::copy_nonoverlapping(
427                        phys_to_virt(addr).as_ptr(),
428                        phys_to_virt(new_addr).as_mut_ptr(),
429                        PAGE_SIZE_4K,
430                    )
431                };
432            }
433        }
434        Ok(new_aspace)
435    }
